# Curator 进化 Skill 片段（v6核心）
def evolve_writing_skill(self, current_rules, reviewer_feedback, iteration):
    # 1. 读取当前Skill
    current_skill = self.memory_manager.read_file("skills/write-sop/SOP_WRITING_SKILL_v{}.md".format(self.latest_skill_version))
    
    # 2. 让Curator（用强模型）生成进化版
    evolution_prompt = f"""
当前Skill: {current_skill}
本轮Reviewer毒舌反馈: {reviewer_feedback}
历史rules更新: {current_rules}

请生成SOP_WRITING_SKILL_v{self.latest_skill_version + 1}.md
要求：
- 保留原有结构和铁律
- 根据反馈新增/强化1-2条新规则（例如更严的条款引用、更强的防抄袭）
- 提高成功标准（通过分+0.2）
- 输出完整Markdown文件内容
"""
    new_skill_content = self.strong_llm.invoke(evolution_prompt)  # Claude Opus / GPT-4o
    
    # 3. 保存新版本 + 更新latest
    new_version = self.latest_skill_version + 1
    self.memory_manager.save_skill("write-sop", f"SOP_WRITING_SKILL_v{new_version}.md", new_skill_content)
    self.latest_skill_version = new_version
    
    return new_skill_content
